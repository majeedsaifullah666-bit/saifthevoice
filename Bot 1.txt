import os
import openai
import telebot
from dotenv import load_dotenv

# Load environment variables
load_dotenv()
openai.api_key = os.getenv("OPENAI_API_KEY")
BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")

bot = telebot.TeleBot(BOT_TOKEN)

# Load persona (your voice)
with open("persona.txt", "r") as f:
    saifthevoice_prompt = f.read()

@bot.message_handler(func=lambda message: True)
def reply_message(message):
    user_text = message.text.strip()

    prompt = f"""{saifthevoice_prompt}

The user said: "{user_text}"
Respond in the same tone and energy as saifthevoice.
"""

    try:
        response = openai.ChatCompletion.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "system", "content": saifthevoice_prompt},
                {"role": "user", "content": user_text}
            ],
            max_tokens=120,
            temperature=0.9
        )

        reply = response['choices'][0]['message']['content']
        bot.reply_to(message, reply)

    except Exception as e:
        bot.reply_to(message, "Something went wrong... even pain has limits.")
        print(e)

print("ðŸ”¥ saifthevoice bot is live...")
bot.polling()
